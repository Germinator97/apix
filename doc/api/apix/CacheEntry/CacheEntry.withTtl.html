Implementation

factory CacheEntry.withTtl({
  required String data,
  required int statusCode,
  required Duration ttl,
  String? etag,
  Map<String, String>? headers,
}) {
  final now = DateTime.now();
  return CacheEntry(
    data: data,
    statusCode: statusCode,
    createdAt: now,
    expiresAt: now.add(ttl),
    etag: etag,
    headers: headers,
  );
}
