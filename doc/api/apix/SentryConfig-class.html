SentryConfig

SentryConfig class

Configuration for the Sentry interceptor.

This interceptor is designed to be error-tracking agnostic. It works with Sentry, Crashlytics, or any other service via callbacks.

Example with Sentry:

final config = SentryConfig(
  captureException: (exception, {stackTrace, extra, tags}) async {
    await Sentry.captureException(
      exception,
      stackTrace: stackTrace,
      withScope: (scope) {
        extra?.forEach((key, value) => scope.setExtra(key, value));
        tags?.forEach((key, value) => scope.setTag(key, value));
      },
    );
  },
  addBreadcrumb: (data) {
    Sentry.addBreadcrumb(Breadcrumb(
      message: data['message'] as String?,
      category: data['category'] as String?,
      data: data['data'] as Map<String, dynamic>?,
      level: SentryLevel.info,
    ));
  },
);
